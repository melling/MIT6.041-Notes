\subsection{n-step Transition Probabilities}

\marginnote{(20:10)}

\begin{align*}
    r_{ij} = P(X_n=j|X_0 = i)
\end{align*}

Key recursion
\begin{align*}
    r_{ij}(n) = \sum_{k=1}^m r_{ik}(n-1)r_{kj}, % \foreach_{i,j}
\end{align*}
