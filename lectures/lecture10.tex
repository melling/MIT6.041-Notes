\subsubsection{Continuous Counterpart}

\begin{align*}
f_{X|Y}(x|y) = \frac{f_{X,Y}(x,y)}{f_Y(y)}  = \frac{f_X(x) f_{Y|X}(y|x)}{f_Y(y)}\\
f_Y(y) = \int_X f_X(x) f_{Y|X}(y,x)dx
\end{align*}
